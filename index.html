4. Ajustes del proyecto ⚙ → Agregar app → Web
5. Copia apiKey, projectId y appId abajo
6. Firestore → Reglas → pega y publica:
rules_version='2';
service cloud.firestore {
 match /databases/{db}/documents {
  match /{d=**}{allow read,write:if true;}
